119 inline void
120 Query<With<Include...>,
121 Without<Exclude...> >::UpdateQuery() {
122 auto getEntityLists = [](auto &... componentArrays) {
123 return std::vector<std::unordered_map<Entity, size_t> >{
124 componentArrays->GetEntityList()...
125 };
126 };
127
128 auto entityArrays = std::apply(getEntityLists, includedComponentArrays);
129
130 if (entityTypeIndex != -1) {
131 entityArrays.erase(entityArrays.begin() + entityTypeIndex);
132 }
133
134 // find smallest array, use it as the base
135 size_t smallestIndex = 0;
136 size_t smallestSize = std::numeric_limits<size_t>::max();
137
138 for (int i = 0; i < entityArrays.size(); ++i) {
139 if (entityArrays[i].size() < smallestSize) {
140 smallestSize = entityArrays[i].size();
141 smallestIndex = i;
142 }
143 }
144 // put smallest list at the front
145 std::swap(entityArrays[0], entityArrays[smallestIndex]);
146
147 included.clear();
148
149 for (auto &[entity, _]: entityArrays[0]) {
150 auto inSet = std::all_of(
151 entityArrays.begin() + 1, entityArrays.end(), [entity](auto &map) {
152 return map.find(entity) != map.end();
153 });
154
155 if (inSet) {
156 included.insert(entity);
157 }
158 }
159
160 // for each excluded list, does entity exist inside
161 // if so, remove
162 auto excludedComponentArrays = std::make_tuple(manager.GetComponentArray<Exclude>()...);
163 auto excludedEntityArrays = std::apply(getEntityLists, excludedComponentArrays);
164
165 for (auto iter = included.begin(); iter != included.end();) {
166 auto entity = *iter;
167 auto inSet = std::any_of(
168 excludedEntityArrays.begin(),
169 excludedEntityArrays.end(),
170 [entity](auto &map) { return map.find(entity) != map.end(); });
171
172 if (inSet) {
173 iter = included.erase(iter);
174 } else {
175 ++iter;
176 }
177 }
178 }
